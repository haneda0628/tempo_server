/* データベース選択 */
use cakephp;

CREATE TABLE members (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    firstname VARCHAR(50) NOT NULL,
    lastname VARCHAR(50) NOT NULL,
    address VARCHAR(255),
    age INT UNSIGNED,
    sex INT UNSIGNED,
    phonenum1 VARCHAR(15),
    phonenum2 VARCHAR(15),
    email1 VARCHAR(50),
    email2 VARCHAR(50),
    role VARCHAR(20),
    created DATETIME DEFAULT NULL,
    modified DATETIME DEFAULT NULL
);

CREATE TABLE members_branches (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    member_id INT UNSIGNED,
    branch_id INT UNSIGNED,
    created DATETIME DEFAULT NULL,
    modified DATETIME DEFAULT NULL
); 

CREATE TABLE users (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50),
    password VARCHAR(255),
    firstname VARCHAR(50),
    lastname VARCHAR(50),
    address VARCHAR(255),
    phonenum VARCHAR(15),
    email VARCHAR(50),
    branch_id INT NOT NULL,
    role VARCHAR(20),
    created DATETIME DEFAULT NULL,
    modified DATETIME DEFAULT NULL
);


CREATE TABLE slides (
	id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	branch_id INT,
	image1 mediumblob,
	image2 mediumblob,
	image3 mediumblob,
	created DATETIME DEFAULT NULL,
	modified DATETIME DEFAULT NULL
);


CREATE TABLE news (
	id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    	user_id INT NOT NULL,
	title VARCHAR(50) NOT NULL,
	contents VARCHAR(1000) NOT NULL,
	image mediumblob,
	created DATETIME DEFAULT NULL,
	modified DATETIME DEFAULT NULL);


CREATE TABLE companies (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50),
    address VARCHAR(255),
    phonenum VARCHAR(15),
    email VARCHAR(50),
    created DATETIME DEFAULT NULL,
    modified DATETIME DEFAULT NULL
);

CREATE TABLE branches (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50),
    company_id INT NOT NULL,
    area_id INT NOT NULL,
    address VARCHAR(255),
    phonenum VARCHAR(15),
    email VARCHAR(50),
    created DATETIME DEFAULT NULL,
    modified DATETIME DEFAULT NULL
);

CREATE TABLE areas (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    postal_code VARCHAR(8),
    prefecture VARCHAR(50),
    city VARCHAR(50),
    town VARCHAR(50),
    created DATETIME DEFAULT NULL,
    modified DATETIME DEFAULT NULL
);

CREATE TABLE messages (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    contents VARCHAR(1000),
    member_id  INT UNSIGNED,
    user_id INT UNSIGNED,
    is_received BOOLEAN,
    message_type INT UNSIGNED,
    created DATETIME DEFAULT NULL,
    modified DATETIME DEFAULT NULL
);

CREATE TABLE coupons (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    branch_id INT UNSIGNED,
    title VARCHAR(50) NOT NULL,
    contents VARCHAR(255) NOT NULL,
    offer_condition VARCHAR(100),
    use_condition VARCHAR(100),
    effectiveness_condition VARCHAR(100),
    expired DATETIME,
    published DATETIME,
    created DATETIME DEFAULT NULL,
    modified DATETIME DEFAULT NULL
);

CREATE TABLE coupons_members (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    coupon_id INT UNSIGNED,
    member_id INT UNSIGNED,
    is_used BOOLEAN,
    created DATETIME DEFAULT NULL,
    modified DATETIME DEFAULT NULL
);

CREATE TABLE stamps (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    branch_id INT UNSIGNED,
    member_id INT UNSIGNED,
    count INT UNSIGNED,
    created DATETIME DEFAULT NULL,
    modified DATETIME DEFAULT NULL
);
